ek_sa_morokei.0004 = {
	type = character_event
	hidden = yes

	immediate = {
		title:d_morokei = {
			holder = {
				save_scope_as = morokei
			}
		}
		title:c_valkygg = {
			holder = {
				top_liege = {
					save_scope_as = war_target
				}
			}
		}
	}

	option = {
		scope:morokei = {
			add_pressed_claim = title:c_valkygg
			start_war = {
				casus_belli = county_conquest_cb
				target = scope:war_target
				target_title = title:c_valkygg
			}
			spawn_army = {
				men_at_arms = {
					stacks = global_var:ek_sa_morokei_progress
					type = risen_dead
				}
				location = title:c_valkygg.title_province
				name = "risen_dead"
				inheritable = no
			}
			if = {
				limit = {
					global_var:ek_sa_morokei_progress > 20
				}
				get_dragon_name = yes
				spawn_army = {
					men_at_arms = {
						type = dragons
						stacks = 1
					}
					inheritable = no
					location = title:c_valkygg.title_province
					save_scope_as = summoned_dragon
					name = ek_religious_army
				}
			}
		}
	}
}

ek_sa_morokei.0007 = {
	type = character_event
	hidden = yes

	immediate = {
		if = {
			limit = {
				scope:sc_loser = scope:morokei
			}
			title:d_morokei = {
				holder = {
					every_character_artifact = {
						set_owner = scope:funder
					}
				}
			}
			destroy_title = title:d_morokei
			scope:morokei = {
				death = {
					killer = scope:victor
					death_reason = death_duel
				}
			}
			set_global_variable = ek_sa_morokei_slain
			scope:funder = {
				trigger_event = ek_sa_morokei.0008
			}
		}
		if = {
			limit = {
				scope:sc_victor = scope:morokei
			}
			scope:sc_loser = {
				death = {
					killer = scope:morokei
					death_reason = death_duel
				}
			}
			if = {
				limit = {
					NOT = {
						scope:sc_loser = scope:funder
					}
				}
				scope:funder = {
					trigger_event = ek_sa_morokei.0009
				}
			}
		}
	}
}
