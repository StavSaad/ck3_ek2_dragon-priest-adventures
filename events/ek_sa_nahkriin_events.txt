	option = {
		scope:nahkriin = {
			add_pressed_claim = title:c_bitterblade
			start_war = {
				casus_belli = county_conquest_cb
				target = scope:war_target
				target_title = title:c_bitterblade
			}
			spawn_army = {
				men_at_arms = {
					stacks = global_var:ek_sa_nahkriin_progress
					type = risen_dead
				}
				location = title:c_bitterblade.title_province
				name = "risen_dead"
				inheritable = no
			}
			if = {
				limit = {
					global_var:ek_sa_nahkriin_progress > 20
				}
				get_dragon_name = yes
				spawn_army = {
					men_at_arms = {
						type = dragons
						stacks = 1
					}
					inheritable = no
					location = title:c_bitterblade.title_province
					save_scope_as = summoned_dragon
					name = ek_religious_army
				}
			}
		}
	}
}

			SC_INITIATOR = root
